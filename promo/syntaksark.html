Dette kan også gjøres med filter via en if-setning. 

from math import sqrt, floor

def er_primtall(n):
    """Gi True hvis n er primtall."""
    if n % 2 == 0 and n != 2:
        return False
    for i in range(3, floor(sqrt(n)) + 1, 2):
        if n % i == 0:
            return False
    if n == 1:
        return False
    return True

primtall = [i for i in range(1, 101) if er_primtall(i)]
print(primtall)
